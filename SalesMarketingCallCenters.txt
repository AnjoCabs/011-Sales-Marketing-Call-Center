CREATE TABLE callcenterdataset
(
	call_ID VARCHAR(50),
	date date,
	Agent_First_Name VARCHAR(50),
	Agent_Last_Name VARCHAR(50),
	Agent_Rating DECIMAL(4,1),
	Product_Discussed VARCHAR(50),
	Call_Duration_Minutes DECIMAL(8,2),
	Call_Outcome VARCHAR(50),
	Customer_Age INT,
	Callers_Name VARCHAR(50),
	Customer_Gender VARCHAR(50),
	State VARCHAR(50),
	Customer_Income_Bracket VARCHAR(50),
	Time_of_Day VARCHAR(50),
	Follow_Up_Call_Required VARCHAR(50),
	Repeat_Customer VARCHAR(50),
	Reason_Call_Abandoned VARCHAR(50));

-- Explaratory Data Analisys ( EDA)

-- 1. How many total calls got?

SELECT
	COUNT(call_ID)
FROM callcenterdataset;

-- 2. Count of Calls  base on the call_outcome

SELECT
	Call_Outcome,
    COUNT(*) AS call_count
FROM callcenterdataset
GROUP BY Call_Outcome;

-- 3. Number of call base on gender

SELECT
	Customer_Gender,
    COUNT(*) AS call_count
FROM callcenterdataset
GROUP BY Customer_Gender;

-- 4. Number of call base on agent

SELECT
	CONCAT( Agent_First_Name, ' ', Agent_Last_Name) AS full_name,
    COUNT(*) AS call_count
FROM callcenterdataset
GROUP BY full_name
ORDER BY call_count DESC;



-- 5. who is the agent that has biggest abandoned calls

SELECT
	CONCAT( Agent_First_Name, ' ', Agent_Last_Name) AS full_name,
    COUNT(Call_Outcome) AS call_count
FROM callcenterdataset
WHERE Call_Outcome = 'Abandoned'
GROUP BY full_name
ORDER BY call_count DESC;


-- 6. percentage of abandoned calls of agent base on the total calls they got

SELECT
    CONCAT(Agent_First_Name, ' ', Agent_Last_Name) AS full_name,
    COUNT(CASE WHEN Call_Outcome = 'Abandoned' THEN 1 END) AS abandoned_calls,
    COUNT(*) AS total_calls,
    ROUND((COUNT(CASE WHEN Call_Outcome = 'Abandoned' THEN 1 END) * 100.0) / COUNT(*),2) AS abandoned_percentage
FROM callcenterdataset
GROUP BY full_name
ORDER BY abandoned_percentage DESC;

-- 7. agent that has highest success calls

SELECT
	CONCAT(Agent_First_Name, ' ', Agent_Last_Name) AS full_name,
    COUNT(CASE WHEN Call_Outcome = 'Success' THEN 1 END) AS success_calls,
	COUNT(*) AS total_calls,
    ROUND((COUNT(CASE WHEN Call_Outcome = 'Success' THEN 1 END) * 100.0) / COUNT(*),2) AS Success_percentage
FROM callcenterdataset
GROUP BY full_name
ORDER BY success_calls DESC;

-- 8. Who has the highest calls base on time

SELECT
	CONCAT(Agent_First_Name, ' ', Agent_Last_Name) AS full_name,
    SUM(Call_Duration_Minutes) AS total_time
FROM callcenterdataset
GROUP BY full_name
ORDER BY total_time DESC;

-- 9.  What is the most common why calls abandoned

SELECT
	Reason_Call_Abandoned,
    COUNT(*) AS count_per_reason
FROM callcenterdataset
GROUP BY Reason_Call_Abandoned
ORDER BY count_per_reason DESC;

-- 10. What time of the day has the highest number of call

SELECT
	Time_of_Day,
    COUNT(*) AS count_
FROM callcenterdataset
GROUP BY Time_of_Day
ORDER BY count_ DESC;

-- 11. What state has the highest abandoned and success

SELECT
	state,
    COUNT(CASE WHEN Call_Outcome = 'Success' THEN 1 END) AS success_call,
    COUNT(CASE WHEN Call_Outcome = 'Failure' THEN 1 END) AS failure_call,
    COUNT(CASE WHEN Call_Outcome = 'Abandoned' THEN 1 END) AS abandoned_call
FROM callcenterdataset
GROUP BY state
ORDER BY
	success_call DESC,
    abandoned_call DESC;

-- 12. What is the common product discussed with success outcome 

SELECT
	Product_Discussed,
    COUNT(CASE WHEN Call_Outcome = 'Success' THEN 1 END) AS count_success
FROM callcenterdataset
GROUP BY  Product_Discussed
ORDER BY count_success DESC;

-- 13. Number of repeat caller

SELECT
	COUNT(*) AS number_repeatCustomer
FROM callcenterdataset
WHERE Repeat_Customer = 'Yes';


-- 14. What is the most common product talked by the repeat customer

SELECT
		Product_Discussed,
        COUNT(CASE WHEN Repeat_Customer = 'Yes' THEN 1 END) AS count
FROM callcenterdataset
GROUP BY product_discussed;
SELECT *
FROM callcenterdataset;

-- 15. what day has the most calls

SELECT
    DAYNAME(date) AS day_ofDate,
    COUNT(*) AS count
FROM callcenterdataset
GROUP BY day_ofDate;
